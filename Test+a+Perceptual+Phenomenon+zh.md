```python
def cal_t(x_mean, y_mean, x_std, y_std, x_count, y_count):
    import numpy as np
    a=x_mean-y_mean
    b=(x_count-1)*x_std*x_std+(y_count-1)*y_std*y_std
    c=x_count+y_count-2
    d=np.sqrt(1/x_count+1/y_count)
    t=a/np.sqrt(b/c)/d
    return t
```
